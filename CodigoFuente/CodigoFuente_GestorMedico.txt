import java.awt.*;
import java.awt.event.*;
import java.time.LocalDate;
import java.time.format.DateTimeFormatter;
import java.time.format.DateTimeParseException;
import java.util.regex.Pattern;
import javax.swing.*;
import javax.swing.table.*;
public class AgendaCitasGUI extends JFrame {
    //Nestro gestor principal
    private GestorCitas gestor;
    // Campos de texto de todo lo que captura el usuario
    private JTextField txtNombre, txtDireccion, txtTelefono,
            txtFechaNac, txtFechaCons, txtDoctor,
            txtMotivo, txtPeso, txtPresion, txtTemperatura;
    // area de texto para diagnostico mas espacio para escribir
    private JTextArea txtDiagnosticoArea;
    private DefaultTableModel modeloTabla;
    private JTable tabla;
    private TableRowSorter<DefaultTableModel> sorter;
    private JButton btnAgregar, btnModificar, btnEliminar, btnBuscar, btnMostrar;
    private JTextField txtBuscar;
    public AgendaCitasGUI() {
        setTitle("Agenda de Citas Medicas - Gestion");
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setSize(1100, 630);
        setLocationRelativeTo(null); // Centra la ventana
        // Creamos el gestor wue carga el archivo solito
        gestor = new GestorCitas();
        initUI();   // Metodo para organizar toda la parte visual
        inicializarTabla(); // Llenamos la tabla con lo que ya habia guardado
    }
    private void inicializarTabla() {
        // Por si acaso ya habia algo lo limpiamos
        modeloTabla.setRowCount(0);
        // Sacamos informacion del gestor y la ponemos en la tabla
        for (Cita c : gestor.getLista()) {
            agregarFilaTabla(c);
        }
    }
    private void agregarFilaTabla(Cita c) {
        // Simplemente mete los datos del objeto a la tabla
        modeloTabla.addRow(new Object[]{
                c.getNombre(), c.getFechaConsulta(), c.getDoctor(), c.getMotivo(), c.getDiagnostico(),
                c.getPeso(), c.getPresion(), c.getTemperatura(), c.getTelefono()
        });
    }
    // METODOS DE ACCION
    private void agregarCita() {
        if (!validarCampos()) return; // Si algo falla cancelamos
        Cita c = crearObjetoDesdeFormulario();
        gestor.agregar(c);     // Se lo damos al gestopara que lo guarde
        agregarFilaTabla(c);   // Lo agamos visualmente
        limpiarCampos();
        JOptionPane.showMessageDialog(this, "Paciente registrado correctamente");
    }
    private void modificarCita() {
        int viewRow = tabla.getSelectedRow();
        if (viewRow == -1) {
            JOptionPane.showMessageDialog(this, "Selecciona una cita para modificar", "Aviso", JOptionPane.INFORMATION_MESSAGE);
            return;
        }
        if (!validarCampos()) return;
        int modelRow = tabla.convertRowIndexToModel(viewRow);
        Cita c = crearObjetoDesdeFormulario();
        gestor.modificar(modelRow, c); // Actualizamos enarchivo
        // Actualizamosvisualmente
        modeloTabla.setValueAt(c.getNombre(), modelRow, 0);
        modeloTabla.setValueAt(c.getFechaConsulta(), modelRow, 1);
        modeloTabla.setValueAt(c.getDoctor(), modelRow, 2);
        modeloTabla.setValueAt(c.getMotivo(), modelRow, 3);
        modeloTabla.setValueAt(c.getDiagnostico(), modelRow, 4);
        modeloTabla.setValueAt(c.getPeso(), modelRow, 5);
        modeloTabla.setValueAt(c.getPresion(), modelRow, 6);
        modeloTabla.setValueAt(c.getTemperatura(), modelRow, 7);
        modeloTabla.setValueAt(c.getTelefono(), modelRow, 8);
        limpiarCampos();
        JOptionPane.showMessageDialog(this, "Modificacion guardada");
    }
    private void eliminarCita() {
        int viewRow = tabla.getSelectedRow();
        if (viewRow == -1) {
            JOptionPane.showMessageDialog(this, "Selecciona una cita para eliminar", "Aviso", JOptionPane.INFORMATION_MESSAGE);
            return;
        }
        // Confirmacion para evitar borrados accidentales
        Object[] opciones = {"Si, eliminar", "No, cancelar"};
        int respuesta = JOptionPane.showOptionDialog(this, "Estas seguro de eliminar este paciente",
                "Confirmar", JOptionPane.YES_NO_OPTION, JOptionPane.QUESTION_MESSAGE, null, opciones, opciones[1]);
        if (respuesta == JOptionPane.YES_OPTION) {
            int modelRow = tabla.convertRowIndexToModel(viewRow);
            gestor.eliminar(modelRow); // Lo borramos de la lista y archivo
            modeloTabla.removeRow(modelRow);
            limpiarCampos();
            JOptionPane.showMessageDialog(this, "Paciente eliminado");
        }
    }
    // METODOS DE APOYO Y VALIDACION
    private Cita crearObjetoDesdeFormulario() {
        // Basicamente empaqueta todos los campos en un objeto Cita
        return new Cita(
                txtNombre.getText().trim(), txtDireccion.getText().trim(), txtTelefono.getText().trim(),
                txtFechaNac.getText().trim(), txtFechaCons.getText().trim(), txtDoctor.getText().trim(),
                txtMotivo.getText().trim(), txtDiagnosticoArea.getText().trim(), txtPeso.getText().trim(),
                txtPresion.getText().trim(), txtTemperatura.getText().trim()
        );
    }
    private boolean validarCampos() {
        // Validacion basica para evitar datos raros o vacios
        if (txtNombre.getText().trim().isEmpty()) {
            JOptionPane.showMessageDialog(this, "El nombre es obligatorio", "Error", JOptionPane.ERROR_MESSAGE);
            return false;
        }
        if (!txtTelefono.getText().trim().matches("\\d{10}")) {
            JOptionPane.showMessageDialog(this, "El telefono debe ser de 10 digitos numericos", "Telefono", JOptionPane.WARNING_MESSAGE);
            return false;
        }
        // Validamos fechas
        String errNac = validarLogicaFecha(txtFechaNac.getText().trim(), true);
        if (errNac != null) {
            JOptionPane.showMessageDialog(this, "Nacimiento: " + errNac, "Fecha", JOptionPane.WARNING_MESSAGE);
            return false;
        }
        String errCons = validarLogicaFecha(txtFechaCons.getText().trim(), false);
        if (errCons != null) {
            JOptionPane.showMessageDialog(this, "Consulta: " + errCons, "Fecha", JOptionPane.WARNING_MESSAGE);
            return false;
        }
        // Validacion de presion
        String presion = txtPresion.getText().trim();
        if (!presion.isEmpty() && !presion.matches("[0-9/]+")) {
            JOptionPane.showMessageDialog(this, "Presion invalida (ej: 120/80).", "Formato", JOptionPane.WARNING_MESSAGE);
            return false;
        }
        // Validacion numerica de peso y temperatura
        try {
            String p = txtPeso.getText().trim();
            String t = txtTemperatura.getText().trim();
            if (!p.isEmpty()) {
                double dp = Double.parseDouble(p);
                if (dp <= 0 || dp > 600) throw new NumberFormatException();
            }
            if (!t.isEmpty()) {
                double dt = Double.parseDouble(t);
                if (dt < 30 || dt > 45) throw new NumberFormatException();
            }
        } catch (NumberFormatException e) {
            JOptionPane.showMessageDialog(this, "Peso o Temperatura irreales/invalidos.", "Error Numerico", JOptionPane.ERROR_MESSAGE);
            return false;
        }
        return true;
    }
    private String validarLogicaFecha(String fechaStr, boolean esNac) {
        try {
            LocalDate f = LocalDate.parse(fechaStr, DateTimeFormatter.ofPattern("dd/MM/yyyy"));
            if (esNac && !f.isBefore(LocalDate.now().plusDays(1))) return "No puede ser futura";
            if (!esNac && f.isBefore(LocalDate.now().minusDays(1))) return "No puede ser pasado";
        } catch (DateTimeParseException e) {
            return "Formato invalido (dd/mm/aaaa)";
        }
        return null;
    }
    private void configurarBoton(JButton boton, Color color) {
        // Estilo al boton
        boton.setBackground(color);
        boton.setForeground(Color.WHITE);
        boton.setFont(new Font("Segoe UI", Font.BOLD, 13));
    }
    private void buscarPorNombre() {
        // Filtro en tiempo real
        String t = txtBuscar.getText().trim();
        if (t.isEmpty()) return;
        sorter.setRowFilter(RowFilter.regexFilter("(?i)" + Pattern.quote(t), 0));
    }
    private void mostrarTodas() {
        // Quita el filtro y limpia la caja
        sorter.setRowFilter(null);
        txtBuscar.setText("");
    }
    private void limpiarCampos() {
        // Limpia todo para evitar confusiones
        txtNombre.setText("");
        txtDireccion.setText("");
        txtTelefono.setText("");
        txtFechaNac.setText("");
        txtFechaCons.setText("");
        txtDoctor.setText("");
        txtMotivo.setText("");
        txtDiagnosticoArea.setText("");
        txtPeso.setText("");
        txtPresion.setText("");
        txtTemperatura.setText("");
    }
    //DISEÑO

        gbcForm.gridy = 1;
        gbcForm.gridwidth = 2;
        formPanel.add(botonesPanel, gbcForm);
        // TABLA DE CITAS
        String[] cols = {"Nombre", "Fecha", "Doctor", "Motivo", "Diagnostico", "Peso", "Presion", "Temp", "Tel"};
        modeloTabla = new DefaultTableModel(cols, 0) {
            public boolean isCellEditable(int r, int c) { return false; }
        };
        tabla = new JTable(modeloTabla);
        tabla.setRowHeight(26);
        sorter = new TableRowSorter<>(modeloTabla);
        tabla.setRowSorter(sorter);
        tabla.addMouseListener(new MouseAdapter() {
            public void mouseClicked(MouseEvent e) {
                if (e.getClickCount() >= 1) {
                    int mr = tabla.convertRowIndexToModel(tabla.getSelectedRow());
                    Cita c = gestor.obtenerCita(mr);
                    cargarForm(c);
                }
            }
        });
        // Panel de bUsqueda
        JPanel busq = new JPanel(new FlowLayout(FlowLayout.LEFT));
        busq.add(new JLabel("Buscar:"));
        busq.add(txtBuscar = new JTextField(20));
        busq.add(crearBoton("Buscar", new Color(90, 170, 90), e -> buscarPorNombre()));

        tabs.addTab("Registro", formPanel);

        pTabla.add(busq, BorderLayout.NORTH);
        pTabla.add(new JScrollPane(tabla), BorderLayout.CENTER);
        tabs.addTab("Citas", pTabla);
        panel.add(tabs, BorderLayout.CENTER);
        add(panel);
    }
    private void addCampo(JPanel p, GridBagConstraints g, int x, int y, String l, Component c) {
        g.gridx = x;
        g.gridy = y;
        g.gridwidth = 1;
        p.add(new JLabel(l), g);
        g.gridx = x + 1;
        p.add(c, g);
    }
    private JButton crearBoton(String t, Color c, ActionListener a) {
        JButton b = new JButton(t);
        configurarBoton(b, c);
        b.addActionListener(a);
        return b;
    }
    private void cargarForm(Cita c) {
        // Carga los datos de la cita seleccionada al formulario
        txtNombre.setText(c.getNombre());
        txtDireccion.setText(c.getDireccion());
        txtTelefono.setText(c.getTelefono());
        txtFechaNac.setText(c.getFechaNacimiento());
        txtFechaCons.setText(c.getFechaConsulta());
        txtDoctor.setText(c.getDoctor());
        txtMotivo.setText(c.getMotivo());
        txtDiagnosticoArea.setText(c.getDiagnostico());
        txtPeso.setText(c.getPeso());
        txtPresion.setText(c.getPresion());
        txtTemperatura.setText(c.getTemperatura());
    }
}

import java.io.Serializable;
public class Cita implements Serializable {
private static final long serialVersionUID = 1L; // Para guardar y leer
    // Todos los datos que representan a una cita/paciente
    private String nombre;
    private String direccion;
    private String telefono;
    private String fechaNacimiento;
    private String fechaConsulta;
    private String doctor;
    private String motivo;
    private String diagnostico;
    private String peso;
    private String presion;
    private String temperatura;
    // Constructor con todo para crear objetos
    public Cita(String nombre, String direccion, String telefono,
        String fechaNacimiento, String fechaConsulta, String doctor,
        String motivo, String diagnostico, String peso,
        String presion, String temperatura) {
        this.nombre = nombre;
        this.direccion = direccion;
        this.telefono = telefono;
        this.fechaNacimiento = fechaNacimiento;
        this.fechaConsulta = fechaConsulta;
        this.doctor = doctor;
        this.motivo = motivo;
        this.diagnostico = diagnostico;
        this.peso = peso;
        this.presion = presion;
        this.temperatura = temperatura;
    }
    // Getters solo devuelven el dato
    public String getNombre() { return nombre; }
    public String getDireccion() { return direccion; }
    public String getTelefono() { return telefono; }
    public String getFechaNacimiento() { return fechaNacimiento; }
    public String getFechaConsulta() { return fechaConsulta; }
    public String getDoctor() { return doctor; }
    public String getMotivo() { return motivo; }
    public String getDiagnostico() { return diagnostico; }
    public String getPeso() { return peso; }
    public String getPresion() { return presion; }
    public String getTemperatura() { return temperatura; }
    // Setters por si luego ocupo editar directamente los atributos
    public void setNombre(String nombre) { this.nombre = nombre; }
    public void setDireccion(String direccion) { this.direccion = direccion; }
    public void setTelefono(String telefono) { this.telefono = telefono; }
    public void setFechaNacimiento(String fechaNacimiento) { this.fechaNacimiento = fechaNacimiento; }
    public void setFechaConsulta(String fechaConsulta) { this.fechaConsulta = fechaConsulta; }
    public void setDoctor(String doctor) { this.doctor = doctor; }
    public void setMotivo(String motivo) { this.motivo = motivo; }
    public void setDiagnostico(String diagnostico) { this.diagnostico = diagnostico; }
    public void setPeso(String peso) { this.peso = peso; }
    public void setPresion(String presion) { this.presion = presion; }
    public void setTemperatura(String temperatura) { this.temperatura = temperatura; }
}

//LOGICA
import java.io.*;
import java.util.ArrayList;
public class GestorCitas {
    private ArrayList<Cita> listaCitas; // Aqui esta todo lo guardado
    private final String NOMBRE_ARCHIVO = "citas_medicas.dat"; // Archivo donde guardamos todo
    public GestorCitas() {
        listaCitas = new ArrayList<>();
        cargarDesdeArchivo(); // Intentamos leer citas si ya existían
    }
    // Devuelve la lista completa la GUI usa esto para la tabla
    public ArrayList<Cita> getLista() {
        return listaCitas;
    }
    public void agregar(Cita c) {
        listaCitas.add(c);
        guardarEnArchivo(); // Guardamos de inmediato
    }
    public void modificar(int indice, Cita c) {
        listaCitas.set(indice, c);
        guardarEnArchivo(); // Guardado automtico para no perder cambios
    }
    public void eliminar(int indice) {
        listaCitas.remove(indice);
        guardarEnArchivo(); // guardamos en archivo
    }
    public Cita obtenerCita(int indice) {
        return listaCitas.get(indice);
    }
    // MÉTODOS PRIVADOS PARA ARCHIVO
    private void guardarEnArchivo() {
        // Guardamos toda la lista en un archivo binario
        try (ObjectOutputStream out = new ObjectOutputStream(new FileOutputStream(NOMBRE_ARCHIVO))) {
            out.writeObject(listaCitas);
        } catch (IOException e) {
            System.err.println("Error al guardar: " + e.getMessage());
        }
    }
    @SuppressWarnings("unchecked")
    private void cargarDesdeArchivo() {
        // Intentamos abrir el archivo de citas previas
        try (ObjectInputStream in = new ObjectInputStream(new FileInputStream(NOMBRE_ARCHIVO))) {
            listaCitas = (ArrayList<Cita>) in.readObject();
        } catch (FileNotFoundException e) {
            // Si no existe el archivo, significa que es la primera vez que se usa
            listaCitas = new ArrayList<>();
        } catch (IOException | ClassNotFoundException e) {
            System.err.println("Error al cargar: " + e.getMessage());
            listaCitas = new ArrayList<>();
        }
    }
}
